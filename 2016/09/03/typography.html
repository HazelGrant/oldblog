    while reality = true

      unless self.sympathies[:need]
        self.easy_come, self.easy_go = true
        self.high, self.low = 0.1
      end

      Piano.play
    end
  end

  private

  def look_up_to_the_skies
    character.eyes.each do |eye|
      eye.open
      eye.look(direction: 'sky')
      eye.see
    end
  end

  def direction(wind)

    when 'north', 'south', 'east', 'west'
      false
    end
  end

  def easy_come_easy_go
    self.come_difficulty == Difficulty::EASY &&
      self.go_difficulty == Difficulty::EASY
  end

    self.high < 0.1 && self.low < 0.1
  end
